using System;

class Program
{
    static void Main()
    {
        bool playAgain = true;

        while (playAgain)
        {
            Console.WriteLine("숫자 맞추기 게임에 오신 것을 환영합니다!");
            Console.WriteLine("1부터 96 사이의 숫자를 맞춰보세요.");

            // 랜덤한 숫자 생성
            Random random = new Random();
            int correctNumber = random.Next(1, 97);

            int attempts = 0;
            int maxAttempts = 10; // 힌트를 줄 시도 횟수

            bool guessed = false;

            while (!guessed && attempts < maxAttempts)
            {
                Console.Write("추측한 숫자를 입력하세요: ");
                string userInput = Console.ReadLine();

                if (!int.TryParse(userInput, out int userGuess))
                {
                    Console.WriteLine("유효한 숫자를 입력하세요.");
                    continue;
                }

                attempts++;

                if (userGuess < correctNumber)
                {
                    Console.WriteLine("너무 낮습니다.");
                }
                else if (userGuess > correctNumber)
                {
                    Console.WriteLine("너무 높습니다.");
                }
                else
                {
                    Console.WriteLine("축하합니다! 숫자를 맞추셨습니다.");
                    guessed = true;
                }

                // 힌트 제공
                if (attempts % 7 == 0 && !guessed)
                {
                    Console.WriteLine($"힌트: 정답은 {correctNumber} 입니다.");
                }
            }

            if (!guessed)
            {
                Console.WriteLine($"정답은 {correctNumber} 였습니다. 다음에 다시 도전해보세요!");
            }

            Console.Write("게임을 다시 하시겠습니까? (y/n): ");
            string playChoice = Console.ReadLine();

            playAgain = (playChoice.ToLower() == "y");
        }

        Console.WriteLine("게임 종료!");
        Console.ReadKey();
    }
}
